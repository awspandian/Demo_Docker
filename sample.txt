1.yml

---
- name: Transfer and execute a script.
  hosts: dev
  user: ansible
  become: yes
  tasks:
     - name: Transfer the script
       copy: src=demo.sh dest=/home/ansible owner: ansible group: ansible mode=0777

     - name: Execute the script
       command: sh ./demo.sh
	   
   	 - name: Change file ownership for script
       path: /home/ansible
	   
	 - name: Remove the Script
       command: sh rm /home/ansible/demo.sh
	   
demo.sh
#!/bin/bash
cd /opt
sudo touch sample.txt
sudo chmod +655 sample.txt
sudo chown ansible:ansible sample.txt

---
- name: Transfer and execute a script.
  hosts: dev
  user: ansible
  become: yes
  tasks:
- name: Transfer the script
copy: src=demo.sh dest=/home/ansible owner=ansible group=ansible mode=0777

- name: Execute the script
command: sh /home/ansible/demo.sh

- name: Remove the Script
ansible.builtin.file:
path: /home/ansible/demo.sh
state: absent
